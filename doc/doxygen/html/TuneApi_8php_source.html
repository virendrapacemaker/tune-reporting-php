62  /**
63  * This function will handle the autoloading of the Tune namespaced
64  * classes.
65  *
66  * @param string $className The name of the class (with prepended namespace) to load.
67  */
68  private function autoloadTuneSdk($className)
69  {
70  // echo 'Trying to load class ', $className, ' via ', __METHOD__, "()\n";
71  if (!class_exists($className)) {
72  // The namespaces map 1-to-1 with the filepaths, so we can just so a
73  // straight conversion.
74  $dirname = dirname(__FILE__);
75  $file = $dirname . '/' . str_replace('\\', '/', $className) . '.php';
76 
77  // echo "file {$file}\n";
78  if (!file_exists($file)) {
79  return false;
80  }
81 
82  // do the actual require now that we've converted it into a file path
83  include_once $file;
84  }
85  }
